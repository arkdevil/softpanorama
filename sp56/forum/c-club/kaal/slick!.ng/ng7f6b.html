   void far _myfree(void);

   Deallocates memory block allocated with _mymalloc(). It's direct port
 to DOS's malloc.

 Parameters :

 None.

 Return :

 None.

 Example :

 #include "window.h"
 #include <stdio.h>

 void main(void)
 {
 char far *s;
     s=_mymalloc(1);
     printf("Got pointer %lx",s);
     if (s)
         _myfree(s);
 }
