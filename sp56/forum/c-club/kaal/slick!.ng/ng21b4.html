   int far locate(int x,int y);

   Position hardware cursor, does not affect cursor size. No checks
 made to validate coordinates.

 Parameters :

 x,y        Viewport relative coordinates.

 Return :

 Always 1.

 Example :

 #include "window.h"
 #include <bios.h>
 #include <dos.h>

 void main(void)
 {
 int i;
     makewnd(0,0,79,3,SINGLEFRAME,0x0a0a,80);
     for (i=0;i<78;i++) {
       locate(i,0);
       delay(300);
     }
     for (i=77;i>=0;i--) {
       locate(i,1);
       delay(300);
     }
     closewnd();
 }
