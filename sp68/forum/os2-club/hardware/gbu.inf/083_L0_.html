Save the following as the file AUD.BAT, edit to change the paths and default IRQ as appropriate, and execute as AUD 5 (for example) to change the SoundBlaster to use IRQ5 under DOS. 

@echo off
if '2' == '%1%' goto valid
if '5' == '%1%' goto valid
if '7' == '%1%' goto valid
if '10' == '%1%' goto valid
SET BLASTER=A220 I10 D1 H5 P330 T6
goto continue
:valid
SET BLASTER=A220 I%1 D1 H5 P330 T6
:continue
SET SOUND=D:\SB16
D:\SB16\SB16SET /R /Q /M:220 /VOC:220 /CD:220 /MIDI:220 /LINE:220 /TREBLE:0
D:\SB16\SBCONFIG.EXE /S
echo BLASTER=%BLASTER%
